	\section{Revisão LFA}
	\subsection{Autômatos Finitos Determinísticos}
	\begin{frame}{Autômatos Finitos Determinísticos}
		Um {\bf autômato finito determinístico} (AFD) é uma 5-upla $(Q, \Sigma, \delta, q_0, F)$, de forma que

		\begin{enumerate}
			\item $Q$ é um conjunto finito conhecido como os {\bf estados},
			\item $\Sigma$ é um conjunto finito chamado o {\bf alfabeto},
			\item $\delta : Q \times \Sigma \rightarrow Q$ é a {\bf função de transição},
			\item $q_0 \in Q$ é o {\bf estado inicial}, e
			\item $F \subseteq Q$ é o {\bf conjunto de estados de aceitação}.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Computação e Linguagem Regular}
		\begin{block}{Computação}
			Seja $M$ um autômato finito e $w = w_1 w_2 \ldots w_n$ seja uma cadeia em que $w_i$ é um membro do alfabeto $\Sigma$. Então $M$ {\bf aceita} $w$ se existe uma sequência de estados $r_0, r_1, \ldots, r_n$ em $Q$ com três condições:
			\begin{enumerate}
				\item $r_0 = q_0$
				\item $\delta(r_i, w_{i+1}) = r_{i+1}$, para $i=0, 1, \ldots, n-1$, e
				\item $r_n \in F$.
			\end{enumerate}
		\end{block}				
		\begin{block}{Linguagem Regular (Definição 1.16)}
			Uma linguagem é chamada de uma {\bf linguagem regular} se algum autômato finito a reconhece.
		\end{block}
	\end{frame}
	
	\begin{frame}{Operações Regulares}
		Sejam $A$ e $B$ linguagens. Definimos as operações regulares {\bf união}, {\bf concatenação} e {\bf estrela} da seguinte forma:
		\begin{itemize}
			\item {\bf União}: $A \cup B = \{ x \mbox{ | } x \in A \mbox{ ou } x \in B \}$.
			\item {\bf Concatenação}: $A \circ B = \{ xy \mbox{ | } x \in A \mbox{ e } y \in B \}$.
			\item {\bf Estrela}: $A^* = \{ x_1 x_2 \ldots x_k \mbox{ | } k \geq 0 \mbox{ e } x_i \in A \}$.
		\end{itemize} \pause
		\begin{block}{Teorema 1.25}
			A classe de linguagens regulares é {\bf fechada} sob a operação de união.
		\end{block}
	\end{frame}
	
	\subsection{Autômato Finito Não-Determinístico}
	\begin{frame}{Autômato Finito Não-Determinístico}
		Um {\bf autômato finito não-determinístico} (AFN) é uma 5-upla $(Q, \Sigma, \delta, q_0, F)$, de forma que

		\begin{enumerate}
			\item $Q$ é um conjunto finito estados,
			\item $\Sigma$ é um alfabeto finito,
			\item $\delta : Q \times \Sigma_\epsilon \rightarrow \mathcal{P}(Q)$ é a função de transição,
			\item $q_0 \in Q$ é o estado inicial, e
			\item $F \subseteq Q$ é o conjunto de estados de aceitação.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Autômato Finito Não-Determinístico}
		\begin{block}{Computação em um AFN}
			Seja $N$ um autômato finito não-determinístico e $w$ uma cadeia sobre o alfabeto $\Sigma$. Então $N$ {\bf aceita} $w$ se podemos escrever $w$ como $w = y_1 y_2 \ldots y_m$, em que cada $y_i$ é um membro de $\Sigma_{\epsilon}$ e existe uma sequência de estados $r_0, r_1, \ldots, r_n$ em $Q$ com três condições:
			\begin{enumerate}
				\item $r_0 = q_0$
				\item $r_{i+1} \in \delta(r_i, y_{i+1})$, para $i=0, 1, \ldots, m-1$, e
				\item $r_m \in F$.
			\end{enumerate}
		\end{block}

	\subsection{Expressões Regulares}
	\begin{frame}{Expressões Regulares}
		Digamos que $R$ é uma {\bf expressão regular} (ER) se $R$ for:

		\begin{enumerate}
			\item $a$, para algum $a \in \Sigma$,
			\item $\epsilon$,
			\item $\emptyset$,
			\item $(R_1 \cup R_2)$, em que $R_1$ e $R_2$ são expressões regulares,
			\item $(R_1 \circ R_2)$, em que $R_1$ e $R_2$ são expressões regulares,
			\item $(R_1^*)$, em que $R_1$ é uma expressão regular.
		\end{enumerate}
	\end{frame}
	
	\begin{frame}{Exemplos de ER}
		\begin{itemize}
			\item $0^* 1 0^*$
			\item $\Sigma^* 1 \Sigma^*$
			\item $\Sigma^* 001 \Sigma^*$
			\item $1^* (01^+)^*$
			\item $(\Sigma \Sigma)^*$
			\item $(0 \cup \epsilon) 1^* = 01^* \cup 1^*$
			\item $1^* \emptyset = \emptyset$
			\item $\emptyset^* = \{ \epsilon \}$
		\end{itemize}
	\end{frame}
	
	\begin{frame}{Expressões Regulares}
		\begin{block}{Teorema}
			Uma linguagem é regular se e somente se alguma expressão regular a descreve.
		\end{block} \pause
		\begin{block}{Estratégia}
			Utilizar para realizar a prova um {\bf autômato finito não-determinístico generalizado}.
		\end{block}
	\end{frame}	
	
	\begin{frame}{Autômato Finito Não-Determinístico Generalizado}
		Um {\bf autômato finito não-determinístico generalizado} (AFNG) é uma 5-upla $(Q, \Sigma, \delta, q_{inicio}, q_{aceita})$, de forma que

		\begin{enumerate}
			\item $Q$ é um conjunto finito estados,
			\item $\Sigma$ é um alfabeto finito,
			\item $\delta : (Q - \{q_{aceita}\}) \times (Q - \{q_{inicio}\}) \rightarrow R$ é a função de transição,
			\item $q_{inicio} \in Q$ é o estado inicial, e
			\item $q_{aceita} \in Q$ é o estado de aceitação.
		\end{enumerate}
	\end{frame}

	\begin{frame}{Linguagens Não-Regulares}
		\begin{alertblock}{}
			\begin{center}
				Existem linguagens que não são regulares como $A = \{0^n 1^n \mbox{ | } n \geq 0 \}$.
			\end{center}
		\end{alertblock} \pause
		\begin{block}{Lema do Bombeamento}
			Se $A$ é uma linguagem regular, então existe um número $p$ (o comprimento do bombeamento) tal que, se $s$ é qualquer cadeia de $A$ de comprimento no mínimo $p$, então $s$ pode ser dividida em três partes, $s = xyz$, satisfazendo as seguintes condições:
			\begin{enumerate}
				\item para cada $i \geq 0, xy^i z \in A$,
				\item $|y| > 0$, e
				\item $|xy| \leq p$.
			\end{enumerate}
		\end{block}
	\end{frame}
